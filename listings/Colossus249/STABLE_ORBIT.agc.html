022296,000044:                                                                                                 #         P38 AND P78 CALCULATE THE REQUIRED DELTA V AND OTHER INITIAL
022297,000045:                                                                                                 #         CONDITIONS REQUIRED BY TEH AGC TO (1) PUT THE ACTIVE VEHICLE
022298,000046:                                                                                                 #         ON A TRANSFER TRAJECTORY THAT INTERCEPTS THE PASSIVE VEHICLE
022299,000047:                                                                                                 #         ORBIT A GIVEN DISTANCE, DELTA R, EITHER AHEAD OF OR BEHIND THE
022300,000048:                                                                                                 #         PASSIVE VEHICLE AND (2) ACTUALLY PLACE THE ACTIVE VEHICLE IN THE
022301,000049:                                                                                                 #         PASSIVE VEHICLE ORBIT WITH A DELTA R SEPARATION BETWEEN THE TWO
022302,000050:                                                                                                 #         VEHICLES
022303,000051: 
022304,000052:                                                                                                 #  CALLING SEQUENCE
022305,000053: 
022306,000054:                                                                                                 #         ASTRONAUT REQUEST THRU DSKY

022309,000057:                                                                                                 #         V37E78E         IF OTHER VEHICLE IS ACTIVE VEHICLE

022312,000060: 
022313,000061:                                                                                                 #         (1)     SOI MANEUVER
022314,000062: 
022315,000063:                                                                                                 #                 (A)  TIG        TIME OF SOI MANEUVER
022316,000064:                                                                                                 #                 (B)  CENTANG    ORBITAL CENTRAL ANGLE OF THE PASSIVE VEHICLE
022317,000065:                                                                                                 #                                 DURING THE TRANSFER FROM TIG TO TIME OF INTERCEPT
022318,000066:                                                                                                 #                 (C)  DELTAR     THE DESIRED SEPARATION OF THE TWO VEHICLES
022319,000067:                                                                                                 #                                 SPECIFIED AS A DISTANCE ALONG THE PASSIVE VEHICLE
022320,000068:                                                                                                 #                                 ORBIT
022321,000069:                                                                                                 #                 (D)  OPTION     EQUALS 1 FOR SOI
022322,000070: 
022323,000071:                                                                                                 #         (2)     SOR MANEUVER
022324,000072: 
022325,000073:                                                                                                 #                 (A)  TIG        TIME OF SOR MANEUVER
022326,000074:                                                                                                 #                 (B)  CENTANG    AN OPTIONAL RESPECIFICATION OF 1 (B) ABOVE
022327,000075:                                                                                                 #                 (C)  OPTION     EQUALS -2 FOR SOR
022328,000076:                                                                                                 #                 (D)  DELTTIME   THE TIME REQUIRED TO TRAVERSE DELTA R WHEN
022329,000077:                                                                                                 #                                 TRAVELING AT A VELOCITY EQUAL TO THE HORIZONTAL
022330,000078:                                                                                                 #                                 VELOCITY OF THE PASSIVE VEHICLE -- SAVED FROM
022331,000079:                                                                                                 #                                 SOI PHASE
022332,000080:                                                                                                 #                 (E)  TINT       TIME OF INTERCEPT (SOI) -- SAVED FROM SOI PHASE

022341,000089:                                                                                                 #                         (FOR SOI ONLY)

022344,000092:                                                                                                 #         (6)  DELVTPI    MAGNITUDE OF DELTA V AT SOI (SOR) TIME
