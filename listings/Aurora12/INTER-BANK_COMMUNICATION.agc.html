001380,000096:                                                                                                  #  THE FOLLOWING SUBROUTINES PROVIDE TO THE BASIC PROGRAMMER ENTRY INTO AND RETURN FROM ANY INTERPRETIVE
001381,000097:                                                                                                  #  CODING WHICH DOES NOT USE THE ENTERING CONTENTS OF Q AND WHICH RETURNS VIA DANZIG. C(A) AND C(L) ARE SAVED.
001382,000098: 
001383,000099:                                                                                                  #  USER'S RESPONSIBILITY TO FILL IN ADVANCE THE APPROPRIATE OPERAND AND ADDRESS REGISTERS USED BY THE
001384,000100:                                                                                                  #  INTERPRETIVE CODING SUCH AS MPAC, BUF, ADDRWD, ETC.; AND TO CONFIRM THAT THE INTERPRETIVE CODING MEETS THE
001385,000101:                                                                                                  #  ABOVE RESTRICTIONS WITH RESPECT TO Q AND DANZIG.
001386,000102: 
001387,000103:                                                                                                  #  USEPRET AND USPRCADR MUST NOT BE USED IN INTERRUPT.
001388,000104: 
001389,000105:                                                                                                  #  1. USEPRET ACCESSES INTERPRETIVE CODING WHICH CAN BE ENTERED WITHOUT CHANGING FBANK.
001390,000106:                                                                                                  #     THE CALLING SEQUENCE IS AS FOLLOWS:
001391,000107: 
001392,000108:                                                                                                  #  L             TC              USEPRET
001393,000109:                                                                                                  #  L+1           TC,TCF          INTPRETX                        TC,TCF MEANS TC OR TCF
001394,000110:                                                                                                  #                                                                INTPRETX IS THE INTERPRETIVE CODING
001395,000111:                                                                                                  #                                                                RETURN IS TO L+2
001396,000112: 
001397,000113:                                                                                                  #  2. USPRCADR ACCESSES INTERPRETIVE CODING IN OTHER THAN THE USER'S FBANK. THE CALLING SEQUENCE IS AS FOLLOWS:
001398,000114: 
001399,000115:                                                                                                  #  L             TC              USPRCADR
001400,000116:                                                                                                  #  L+1           CADR            INTPRETX                        INTPRETX IS THE INTERPRETIVE CODING
001401,000117:                                                                                                  #                                                                RETURN IS TO L+2
001402,000118: 
001403,000119:    4131           56002        USEPRET            XCH      Q                                     #  FETCH Q, SAVING A
001404,000120:    4132           54150                           TS       LOC                                   #  L+1 TO LOC
001405,000121:    4133           30004                           CA       FBANK                                 
001406,000122:    4134           54151                           TS       BANKSET                               #  USERS BANK TO BANKSET
001407,000123:    4135           37650                           CA       BIT8                                  
001408,000124:    4136           54023                           TS       EDOP                                  #  EXIT INSTRUCTION TO EDOP
001409,000125:    4137           30002                           CA       Q                                     #  RETRIEVE ORIGINAL A
001410,000126:    4140           00150                           TC       LOC                                   
001411,000127: 
001412,000128:    4141           54150        USPRCADR           TS       LOC                                   #  SAVE A
001413,000129:    4142           37650                           CA       BIT8                                  
001414,000130:    4143           54023                           TS       EDOP                                  #  EXIT INSTRUCTION TO EDOP
001415,000131:    4144           30004                           CA       FBANK                                 
001416,000132:    4145           54151                           TS       BANKSET                               #  USERS BANK TO BANKSET
001417,000133:    4146           50002                           INDEX    Q                                     
001418,000134:    4147           30000                           CA       0                                     
001419,000135:    4150           54004                           TS       FBANK                                 #  INTERPRETIVE BANK TO FBANK
001420,000136:    4151           76426                           MASK     LOW10                                 #  YIELDS INTERPRETIVE RELATIVE ADDRESS
001421,000137:    4152           56002                           XCH      Q                                     #  INTERPRETIVE ADDRESS TO Q, FETCHING L+1
001422,000138:    4153           56150                           XCH      LOC                                   #  L+1 TO LOC, RETRIEVING ORIGINAL A
001423,000139:    4154           50002                           INDEX    Q                                     
001424,000140:    4155           12000                           TCF      10000
