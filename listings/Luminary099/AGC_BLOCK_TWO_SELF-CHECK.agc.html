054807,000031:                                                                                                  #  PROGRAM DESCRIPTION                           DATE   20 DECEMBER 1967
054808,000032:                                                                                                  #  PROGRAM NAME - SELF-CHECK                     LOG SECTION  AGC BLOCK TWO SELF-CHECK
054809,000033:                                                                                                  #  MOD NO - 1                                    ASSEMBLY SUBROUTINE UTILITYM REV 25
054810,000034:                                                                                                  #  MOD BY - GAUNTT
054811,000035: 
054812,000036:                                                                                                  #  FUNCTIONAL DESCRIPTION
054813,000037:                                                                                                  #        PROGRAM HAS TWO MAIN PARTS.  THE FIRST IS SELF-CHECK WHICH RUNS AS A ZERO PRIORITY JOB WITH NO CORE SET, AS
054814,000038:                                                                                                  #        PART OF THE BACK-UP IDLE LOOP.  THE SECOND IS SHOW-BANKSUM WHICH RUNS AS A REGULAR EXECUTIVE JOB WITH ITS OWN
054815,000039:                                                                                                  #        STARTING VERB.
054816,000040: 
054817,000041:                                                                                                  #        THE PURPOSE OF SELF-CHECK IS TO CHECK OUT VARIOUS PARTS OF THE COMPUTER AS OUTLINED BELOW IN THE OPTIONS.
054818,000042: 
054819,000043:                                                                                                  #        THE PURPOSE OF SHOW-BANKSUM IS TO DISPLAY THE SUM OF EACH BANK, ONE AT A TIME.
054820,000044: 
054821,000045:                                                                                                  #        IN ALL THERE ARE 7 POSSIBLE OPTIONS IN THIS BLOCK II VERSION OF SELF-CHECK.  MORE DETAIL DESCRIPTION MAY BE
054822,000046:                                                                                                  #        FOUND IN E-2065 BLOCK II AGC SELF-CHECK AND SHOW BANKSUM BY EDWIN D. SMALLY DECEMBER 1966, AND ADDENDA 2 AND 3.
054823,000047: 
054824,000048:                                                                                                  #        THE DIFFERENT OPTIONS ARE CONTROLLED BY PUTTING DIFFERENT NUMBERS IN THE SMODE REGISTER (NOUN 27).  BELOW IS
054825,000049:                                                                                                  #        A DESCRIPTION OF WHAT PARTS OF THE COMPUTER THAT ARE CHECKED BY THE OPTIONS, AND THE CORRESPONDING NUMBER, IN
054826,000050:                                                                                                  #        OCTAL, TO LOAD INTO SMODE.
054827,000051:                                                                                                  #                +-4             ERASABLE MEMORY
054828,000052:                                                                                                  #                +-5             FIXED MEMORY
054829,000053:                                                                                                  #                +-1,2,3,6,7,10  EVERYTHING IN OPTIONS 4 AND 5.
054830,000054:                                                                                                  #                -0              SAME AS +-10 UNTIL AN ERROR IS DETECTED.
054831,000055:                                                                                                  #                +0              NO CHECK, PUTS COMPUTER INTO THE BACKUP IDLE LOOP.
054832,000056: 
054833,000057:                                                                                                  #  WARNINGS
054834,000058:                                                                                                  #        USE OF E MEMORY RESERVED FOR SELF-CHECK (EVEN IN IDLE LOOP) AS TEMP STORAGE BY OTHER PROGRAMS IS DANGEROUS.
054835,000059:                                                                                                  #        SMODE SET GREATER THAN OCT 10 PUTS COMPUTER INTO BACKUP IDLE LOOP.
054836,000060: 
054837,000061:                                                                                                  #  CALLING SEQUENCE
054838,000062:                                                                                                  #        TO CALL SELF-CHECK KEY IN
054839,000063:                                                                                                  #                V 21 N 27 E     OPTION NUMBER E
054840,000064:                                                                                                  #        TO CALL SHOW-BANKSUM KEY IN
054841,000065:                                                                                                  #                V 91 E          DISPLAYS FIRST BANK
054842,000066:                                                                                                  #                V 33 E          PROCEED, DISPLAYS NEXT BANK
054843,000067: 
054844,000068:                                                                                                  #  EXIT MODES, NORMAL AND ALARM
054845,000069:                                                                                                  #        SELF-CHECK NORMALLY CONTINUES INDEFINITELY UNLESS THERE IS AN ERROR DETECTED.  IF SO + OPTION NUMBERS PUT
054846,000070:                                                                                                  #        COMPUTER INTO BACKUP IDLE LOOP, - OPTION NUMBERS RESTART THE OPTION.
054847,000071: 
054848,000072:                                                                                                  #        THE -0 OPTION PROCEEDS FROM THE LINE FOLLOWING THE LINE WHERE THE ERROR WAS DETECTED.
054849,000073: 
054850,000074:                                                                                                  #        SHOW-BANKSUM PROCEEDS UNTIL A TERMINATE IS KEYED IN (V 34 E).  THE COMPUTER IS PUT INTO THE BACKUP IDLE LOOP
054851,000075: 
054852,000076:                                                                                                  #  OUTPUT

Page 1285 

054854,000078:                                                                                                  #        SELF-CHECK UPON DETECTING AN ERROR LOADS THE SELF-CHECK ALARM CONSTANT (01102) INTO THE FAILREG SET AND
054855,000079:                                                                                                  #        TURNS ON THE ALARM LIGHT.  THE OPERATOR MAY THEN DISPLAY THE THREE FAILREGS BY KEYING IN V 05 N 09 E.  FOR FURTHER
054856,000080:                                                                                                  #        INFORMATION HE MAY KEY IN V 05 N 08 E, THE DSKY DISPLAY IN R1 WILL BE ADDRESS+1 OF WHERE THE ERROR WAS DETECTED,
054857,000081:                                                                                                  #        IN R2 THE BBCON OF SELF-CHECK, AND IN R3 THE TOTAL NUMBER OF ERRORS DETECTED BY SELF-CHECK SINCE THE LAST MAN
054858,000082:                                                                                                  #        INITIATED FRESH START (SLAP1).
054859,000083: 
054860,000084:                                                                                                  #        SHOW-BANKSUM STARTING WTIH BANK 0 DISPLAYS IN R1 THE BANK SUM (A +-NUMBER EQUAL TO THE BANK NUMBER), IN R2
054861,000085:                                                                                                  #        THE BANK NUMBER, AND IN R3 THE BUGGER WORD.
054862,000086: 
054863,000087:                                                                                                  #  ERASABLE INITIALIZATION REQUIRED
054864,000088:                                                                                                  #        ACCOMPLISHED BY FRESH START
054865,000089:                                                                                                  #                SMODE SET TO +0
054866,000090: 
054867,000091:                                                                                                  #  DEBRIS
054868,000092:                                                                                                  #        ALL EXITS FROM THE CHECK OF ERASABLE (ERASCHK) RESTORE ORIGINAL CONTENTS TO REGISTERS UNDER CHECK.
054869,000093:                                                                                                  #        EXCEPTION IS A RESTART.  RESTART THAT OCCURS DURING ERASCHK RESTORES ERASABLE, UNLESS THERE IS EVIDENCE TO DOUBT
054870,000094:                                                                                                  #        E MEMORY, IN WHICH CASE PROGRAM THEN DOES A FRESH START (DOFSTART).
