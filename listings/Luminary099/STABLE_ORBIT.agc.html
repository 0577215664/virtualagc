030913,000038: 
030914,000039:                                                                                                 #         P38 AND P78 CALCULATE THE REQUIRED DELTA V AND OTHER INITIAL
030915,000040:                                                                                                 #         CONDITIONS REQUIRED BY THE AGC TO (1) PUT THE ACTIVE VEHICLE
030916,000041:                                                                                                 #         ON A TRANSFER TRAJECTORY THAT INTERCEPTS THE PASSIVE VEHICLE
030917,000042:                                                                                                 #         ORBIT A GIVEN DISTANCE, DELTA R, EITHER AHEAD OF OR BEHIND THE
030918,000043:                                                                                                 #         PASSIVE VEHICLE AND (2) ACTUALLY PLACE THE ACTIVE VEHICLE IN THE
030919,000044:                                                                                                 #         PASSIVE VEHICLE ORBIT WITH A DELTA R SEPARATION BETWEEN THE TWO
030920,000045:                                                                                                 #         VEHICLES
030921,000046: 
030922,000047:                                                                                                 #  CALLING SEQUENCE
030923,000048: 
030924,000049:                                                                                                 #         ASTRONAUT REQUEST THRU DSKY

030927,000052:                                                                                                 #         V37E78E         IF OTHER VEHICLE IS ACTIVE VEHICLE
030928,000053: 
030929,000054:                                                                                                 #  INPUT
030930,000055: 
030931,000056:                                                                                                 #         (1)     SOI MANEUVER
030932,000057: 
030933,000058:                                                                                                 #                 (A)  TIG        TIME OF SOI MANEUVER
030934,000059:                                                                                                 #                 (B)  CENTANG    ORBITAL CENTRAL ANGLE OF THE PASSIVE VEHICLE
030935,000060:                                                                                                 #                                 DURING THE TRANSFER FROM TIG TO TIME OF INTERCEPT
030936,000061:                                                                                                 #                 (C)  DELTAR     THE DESIRED SEPARATION OF THE TWO VEHICLES
030937,000062:                                                                                                 #                                 SPECIFIED AS A DISTANCE ALONG THE PASSIVE VEHICLE
030938,000063:                                                                                                 #                                 ORBIT
030939,000064:                                                                                                 #                 (D)  OPTION     EQUALS 1 FOR SOI
030940,000065: 
030941,000066:                                                                                                 #         (2)     SOR MANEUVER
030942,000067: 
030943,000068:                                                                                                 #                 (A)  TIG        TIME OF SOR MANEUVER
030944,000069:                                                                                                 #                 (B)  CENTANG    AN OPTIONAL RESPECIFICATION OF 1 (B) ABOVE
030945,000070:                                                                                                 #                 (C)  OPTION     EQUALS 2 FOR SOR
030946,000071:                                                                                                 #                 (D)  DELTTIME   THE TIME REQUIRED TO TRAVERSE DELTA R WHEN
030947,000072:                                                                                                 #                                 TRAVELING AT A VELOCITY EQUAL TO THE HORIZONTAL
030948,000073:                                                                                                 #                                 VELOCITY OF THE PASSIVE VEHICLE -- SAVED FROM
030949,000074:                                                                                                 #                                 SOI PHASE
030950,000075:                                                                                                 #                 (E)  TINT       TIME OF INTERCEPT (SOI) -- SAVED FROM SOI PHASE
030951,000076:

030956,000081:                                                                                                 #         (3)  +MGA       MIDDLE GIMBAL ANGLE
030957,000082:                                                                                                 #  Page 724
030958,000083:                                                                                                 #         (4)  DSPTEM1    TIME OF INTERCEPT OF PASSIVE VEHICLE ORBIT
030959,000084:                                                                                                 #                         (FOR SOI ONLY)
030960,000085:                                                                                                 #         (5)  POSTTPI    PERIGEE ALTITUDE OF ACTIVE VEHICLE ORBIT AFTER
030961,000086:                                                                                                 #                         THE SOI (SOR) MANEUVER
030962,000087:                                                                                                 #         (6)  DELVTPI    MAGNITUDE OF DELTA V AT SOI (SOR) TIME
030963,000088:                                                                                                 #         (7)  DELVTPF    MAGNITUDE OF DELTA V AT INTERCEPT TIME
030964,000089:                                                                                                 #         (8)  DELTA      VELOCITY AT SOI (AND SOR) -- LOCAL VERTICAL
030965,000090:                                                                                                 #                         CORDINATES
