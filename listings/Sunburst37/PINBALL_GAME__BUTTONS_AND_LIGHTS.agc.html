008997,000030:                                                                                                  #  FUNCTIONAL DESCRIPTION-
008998,000031: 
008999,000032:                                                                                                  #     THE KEYBOARD AND DISPLAY SYSTEM PROGRAM OPERATES UNDER EXECUTIVE
009000,000033:                                                                                                  #  CONTROL AND PROCESSES INFORMATION EXCHANGED BETWEEN THE AGC AND THE
009001,000034:                                                                                                  #  COMPUTER OPERATOR.  THE INPUTS TO THE PROGRAM ARE FROM THE KEYBOARD,
009002,000035:                                                                                                  #  FROM INTERNAL PROGRAMS, AND FROM THE UPLINK.
009003,000036:                                                                                                  #     THE LANGUAGE OF COMMUNICATION WITH THE PROGRAM IS A PAIR OF WORDS
009004,000037:                                                                                                  #  KNOWN AS VERB AND NOUN.  EACH OF THESE IS REPRESENTED BY A 2 CHARACTER
009005,000038:                                                                                                  #  OCTAL NUMBER.  THE VERB CODE INDICATES WHAT ACTION IS TO BE TAKEN, 
009006,000039:                                                                                                  #  NOUN CODE INDICATES TO WHAT THIS ACTION IS APPLIED.  NOUNS USUALLY
009007,000040:                                                                                                  #  REFER TO A GROUP OF ERASABLE REGISTERS.
009008,000041: 
009009,000042:                                                                                                  #  VERBS ARE GROUPED INTO DISPLAYS, LOADS, MONITORS (DISPLAYS THAT ARE
009010,000043:                                                                                                  #  UPDATED ONCE PER SECOND), SPECIAL FUNCTIONS, AND EXTENDED VERBS(THESE
009011,000044:                                                                                                  #  ARE OUTSIDE OF THE DOMAIN OF PINBALL AND CAN BE FOUND UNDER LOG SECTION
009012,000045:                                                                                                  #  :EXTENDED VERBS:).
009013,000046:                                                                                                  #  A LIST OF VERBS AND NOUNS IS GIVEN IN LOG SECTION :ASSEMBLY AND 
009014,000047:                                                                                                  #  OPERATION INFORMATION:.
009015,000048: 
009016,000049: 
009017,000050:                                                                                                  #  CALLING SEQUENCES -
009018,000051: 
009019,000052:                                                                                                  #  KEYBOARD:
009020,000053:                                                                                                  #     EACH DEPRESSION OF A KEYBOARD BUTTON ACTIVATES INTERRUPT KEYRUPT1
009021,000054:                                                                                                  #  AND PLACES THE 5 BIT KEY CODE INTO CHANNEL 15.  KEYRUPT1 PLACES THE KEY
009022,000055:                                                                                                  #  CODE INTO MPAC, ENTERS AN EXECUTIVE REQUEST FOR THE KEYBOARD AND DISPLAY
009023,000056:                                                                                                  #  PROGRAM (AT :CHARIN:), AND EXECUTES A RESUME.
009024,000057: 
009025,000058:                                                                                                  #  UPLINK:
009026,000059:                                                                                                  #     EACH WORD RECEIVED BY THE UPLINK ACTIVATES INTERRUPT UPRUPT WHICH
009027,000060:                                                                                                  #  PLACES THE 5 BIT KEY CODE INTO MPAC, ENTERS AN EXECUTIVE REQUEST FOR THE
009028,000061:                                                                                                  #  KEYBOARD AND DISPLAY PROGRAM (AT:CHARIN:) AND EXECUTES A RESUME.
009029,000062: 
009030,000063:                                                                                                  #  INTERNAL PROGRAMS:
009031,000064:                                                                                                  #     INTERNAL PROGRAMS CALL PINBALL AT :NVSUB: WITH THE DESIRED VERB/NOUN
009032,000065:                                                                                                  #  CODE IN A (LOW 6 BITS FOR NOUN, NEXT 6 BITS FOR VERB).  DETAILS
009033,000066:                                                                                                  #  DESCRIBED ON REMARKS CARDS JUST BEFORE :NVSUB: AND :NVSUBWAIT: (SEE
009034,000067: 
009035,000068:                                                                                                  #  SYMBOL TABLE FOR PAGE NUMBERS).
009036,000069:                                                                                                  #     THERE IS AN INTERLOCK SYSTEM FOR INTERNAL PROGRAMS THAT MUST BE USED
009037,000070:                                                                                                  #  IN CONJUNCTION WITH :NVSUB:.  THIS IS CALLED GRAB/FREE.  DETAILS
009038,000071:                                                                                                  #  DESCRIBED ON REMARKS JUST BEFORE :GRABDSP:, :GRABWAIT: (SEE SYMBOL TABLE
009039,000072:                                                                                                  #  FOR PAGE NUMBERS).
