037586,000156:                                                                                                  #  GOTOADDR CONTAINS THE ADDRESS OF THE ROLL COMMAND EQUATIONS APPROPRIATE TO THE CURRENT PHASE OF
037587,000157:                                                                                                  #  RE-ENTRY.  SEQUENCING IS AS FOLLOWS:
037588,000158: 
037589,000159:                                                                                                  #  INITROLL      ADDRESS IS SET HERE INITIALLY. HOLDS INITIAL ROLL ATTITUDE UNTIL KAT  IS EXCEEDED. THEN HOLDS NEW ROLL
037590,000160:                                                                                                  #                ATTITUDE UNTIL  VRTHRESH  IS EXCEEDED.  THEN BRANCHES TO
037591,000161: 
037592,000162:                                                                                                  #  HUNTEST       THIS SECTION CHECKS TO SEE IF THE PREDICTED RANGE AT NOMINAL   L/D FROM PRESENT CONDITIONS IS LESS
037593,000163:                                                                                                  #                THAN THE DESIRED RANGE.
037594,000164:                                                                                                  #                  IF NOT - A ROLL COMMAND IS GENERATED BY THE CONSTANT DRAG CONTROLLER.
037595,000165:                                                                                                  #                  IF SO - CONTROL AND GOTOADDR ARE SET TO  UPCONTRL.
037596,000166:                                                                                                  #                USUALLY NO ITERATION IS INVOLVED EXCEPT IF THE RANGE DESIRED IS TOO LONG ON THE FIRST PASS THROUGH
037597,000167:                                                                                                  #                HUNTEST.
037598,000168: 
037599,000169:                                                                                                  #  UPCONTRL      CONTROLS ROLL DURING THE SUPER-CIRCULAR PHASE.  UPCONTRL IS TERMINATED EITHER
037600,000170:                                                                                                  #                  (A)  WHEN THE DRAG (AS MEASURED BY THE PIPAS) FALLS BELOW Q7, OR
037601,000171:                                                                                                  #                  (B)  IF RDOT IS NEGATIVE AND REFERENCE VL EXCEEDS V.
037602,000172:                                                                                                  #                IN CASE (A),  GOTOADDR  IS SET TO  KEP2  AND IN CASE (B), TO  PREDICT3  SKIPPING THE KEPLER PHASE OF
037603,000173:                                                                                                  #                ENTRY.
037604,000174: 
037605,000175:                                                                                                  #  KEP2          GOTOADDR IS SET HERE DURING THE KEPLER PHASE TO MONITOR DRAG. THE SPACECRAFT IS INSTANTANEOUSLY
037606,000176:                                                                                                  #                TRIMMED IN PITCH AND YAW TO THE COMPUTED RELATIVE VELOCITY. THE LAST COMPUTED ROLL ANGLE IS MAINTAINED.
037607,000177:                                                                                                  #                WHEN THE MEASURED DRAG EXCEEDS Q7 +0.5,  GOTOADDR  IS SET TO
037608,000178: 
037609,000179:                                                                                                  #  PREDICT3      THIS CONTROLS THE FINAL SUB-ORBITAL PHASE. ROLL COMMANDS CEASE
037610,000180:                                                                                                  #                WHEN  V  IS LESS THAN VQUIT. AN EXIT IS MADE TO
037611,000181: 
037612,000182:                                                                                                  #  P67.1         THE LAST COMPUTED ROLL ANGLE IS MAINTAINED. RATE DAMPING IS DONE IN PITCH AND YAW. PRESENT LATITUDE
