032878,000110:                                                                                                 #  DDUMCALC IS PROGRAMMED AS FOLLOWS:
032879,000111:                                                                                                 #                                          2                                           ___
032880,000112:                                                                                                 #               (RIGNZ - RGU )/16 + 16(RGU  )KIGNY/B8 + (RGU - RIGNX)KIGNX/B4 + (ABVAL(VGU) - VIGN)KIGNV/B4
032881,000113:                                                                                                 #                           2             1                 0
032882,000114:                                                                                                 #         DDUM = -------------------------------------------------------------------------------------------
032883,000115:                                                                                                 #                                                 10
032884,000116:                                                                                                 #                                                2   (VGU - 16 VGU KIGNX/B4)
032885,000117:                                                                                                 #                                                        2        0
032886,000118:                                                                                                 #  Page 787
032887,000119:                                                                                                 #  THE NUMERATOR IS SCALED IN METERS AT 2(28).  THE DENOMINATOR IS A VELOCITY IN UNITS OF 2(10) M/CS.
032888,000120:                                                                                                 #  THE QUOTIENT IS THUS A TIME IN UNITS OF 2(18) CENTISECONDS.  THE FINAL SHIFT RESCALES TO UNITS OF 2(28) CS.
032889,000121:                                                                                                 #  THERE IS NO DAMPING FACTOR.  THE CONSTANTS KIGNX/B4, KIGNY/88 AND KIGNV/B4 ARE ALL NEGATIVE IN SIGN.
