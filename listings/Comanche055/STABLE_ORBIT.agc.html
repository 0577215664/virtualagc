022797,000038:                                                                                                 #         P38 AND P78 CALCULATE THE REQUIRED DELTA V AND OTHER INITIAL
022798,000039:                                                                                                 #         CONDITIONS REQUIRED BY THE AGC TO (1) PUT THE ACTIVE VEHICLE
022799,000040:                                                                                                 #         ON A TRANSFER TRAJECTORY THAT INTERCEPTS THE PASSIVE VEHICLE

022801,000042:                                                                                                 #         PASSIVE VEHICLE AND (2) ACTUALLY PLACE THE ACTIVE VEHICLE IN THE
022802,000043:                                                                                                 #         PASSIVE VEHICLE ORBIT WITH A DELTA R SEPARATION BETWEEN THE TWO
022803,000044:                                                                                                 #         VEHICLES
022804,000045: 
022805,000046:                                                                                                 #  CALLING SEQUENCE
022806,000047: 
022807,000048:                                                                                                 #         ASTRONAUT REQUEST THRU DSKY

022810,000051:                                                                                                 #         V37E78E         IF OTHER VEHICLE IS ACTIVE VEHICLE

022813,000054: 
022814,000055:                                                                                                 #         (1)     SOI MANEUVER
022815,000056: 
022816,000057:                                                                                                 #                 (A)  TIG        TIME OF SOI MANEUVER
022817,000058:                                                                                                 #                 (B)  CENTANG    ORBITAL CENTRAL ANGLE OF THE PASSIVE VEHICLE
022818,000059:                                                                                                 #                                 DURING THE TRANSFER FROM TIG TO TIME OF INTERCEPT
022819,000060:                                                                                                 #                 (C)  DELTAR     THE DESIRED SEPARATION OF THE TWO VEHICLES
022820,000061:                                                                                                 #                                 SPECIFIED AS A DISTANCE ALONG THE PASSIVE VEHICLE
022821,000062:                                                                                                 #                                 ORBIT
022822,000063:                                                                                                 #                 (D)  OPTION     EQUALS 1 FOR SOI
022823,000064: 
022824,000065:                                                                                                 #         (2)     SOR MANEUVER
022825,000066: 
022826,000067:                                                                                                 #                 (A)  TIG        TIME OF SOR MANEUVER
022827,000068:                                                                                                 #                 (B)  CENTANG    AN OPTIONAL RESPECIFICATION OF 1 (B) ABOVE
022828,000069:                                                                                                 #                 (C)  OPTION     EQUALS 2 FOR SOR
022829,000070:                                                                                                 #                 (D)  DELTTIME   THE TIME REQUIRED TO TRAVERSE DELTA R WHEN
022830,000071:                                                                                                 #                                 TRAVELING AT A VELOCITY EQUAL TO THE HORIZONTAL
022831,000072:                                                                                                 #                                 VELOCITY OF THE PASSIVE VEHICLE - SAVED FROM
022832,000073:                                                                                                 #                                 SOI PHASE
022833,000074:                                                                                                 #                 (E)  TINT       TIME OF INTERCEPT (SOI) - SAVED FROM SOI PHASE

022844,000085:                                                                                                 #                         THE SOI (SOR) MANEUVER
022845,000086:                                                                                                 #         (6)  DELVTPI    MAGNITUDE OF DELTA V AT SOI (SOR) TIME
022846,000087:                                                                                                 #         (7)  DELVTPF    MAGNITUDE OF DELTA V AT INTERCEPT TIME
