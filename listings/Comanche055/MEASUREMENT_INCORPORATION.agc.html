054182,000031:                                                                                                 #  INCORP1 -- PERFORMS THE SIX DIMENSIONAL STATE VECTOR DEVIATION FOR POSITION
054183,000032:                                                                                                 #  AND VELOCITY OR THE NINE-DIMENSIONAL DEVIATION OF POSITION, VELOCITY, AND
054184,000033:                                                                                                 #  RADAR OR LANDMARK BIAS.  THE OUTPUT OF THE BVECTOR ROUTINE ALONG WITH THE
054185,000034:                                                                                                 #  ERROR TRANSITION MATRIX (W) ARE USED AS INPUT TO THE ROUTINE.  THE DEVIATION
054186,000035:                                                                                                 #  IS OBTAINED BY COMPUTING AN ESTIMATED TRACKING MEASUREMENT FROM THE
054187,000036:                                                                                                 #  CURRENT STATE VECTOR AND COMPARING IT WITH AN ACTUAL TRACKING MEASUREMENT
054188,000037:                                                                                                 #  AND APPLYING A STATISTICAL WEIGHTING VECTOR.

054193,000042:                                                                                                 #         VARIANCE = VARIANCE (SCALAR)
054194,000043:                                                                                                 #           DELTAQ = MEASURED DEVIATION (SCALAR)
054195,000044:                                                                                                 #          BVECTOR = 6 OR 9 DIMENSIONAL BVECTOR
