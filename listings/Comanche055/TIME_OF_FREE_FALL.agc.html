059924,000235:                                                                                                  #                RP = P/(1+E)                            RA = (1+E) / ALFA
059925,000236:                                                                                                  #        WHERE    2
059926,000237:                                                                                                  #                E  = 1 - P ALFA
059927,000238:                                                                                                  #        IF RA IS NEGATIVE OR SHOWS DIVIDE OVERFLOW, THEN RA = POSMAX BECAUSE
059928,000239:                                                                                                  #                1. APOGEE RADIUS IS NOT MEANINGFUL FOR HYPERBOLA
059929,000240:                                                                                                  #                2. APOGEE RADIUS IS NOT DEFINED FOR PARABOLA
059930,000241:                                                                                                  #                3. APOGEE RADIUS EXCEEDS THE SCALING FOR ELLIPSE.
059931,000242:
