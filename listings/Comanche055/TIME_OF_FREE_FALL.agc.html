059606,000227:                                                                                                  #  FUNCTIONAL DESCRIPTION:  USED BY CALCTPER AND TFF DISPLAYS TO CALCULATE PERIGEE RADIUS AND ALSO
059607,000228:                                                                                                  #        APOGEE RADIUS FOR A GENERAL CONIC.
059608,000229:                                                                                                  #        PROGRAM GIVES PERIGEE RADIUS AS         APOGEE RADIUS IS GIVEN BY
059609,000230:                                                                                                  #                RP = P/(1+E)                            RA = (1+E) / ALFA
059610,000231:                                                                                                  #        WHERE    2
059611,000232:                                                                                                  #                E  = 1 - P ALFA
059612,000233:                                                                                                  #        IF RA IS NEGATIVE OR SHOWS DIVIDE OVERFLOW, THEN RA = POSMAX BECAUSE
059613,000234:                                                                                                  #                1. APOGEE RADIUS IS NOT MEANINGFUL FOR HYPERBOLA
059614,000235:                                                                                                  #                2. APOGEE RADIUS IS NOT DEFINED FOR PARABOLA
059615,000236:                                                                                                  #                3. APOGEE RADIUS EXCEEDS THE SCALING FOR ELLIPSE.
