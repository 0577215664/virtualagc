060890,000085:                                                                                                 #  PHASCHNG IS THE MAIN WAY OF MAKING PHASE CHANGES FOR RESTARTS.  THERE ARE THREE FORMS OF PHASCHNG, KNOWN AS TYPE
060891,000086:                                                                                                 #  A, TYPE B, AND TYPE C.  THEY ARE ALL CALLED AS FOLLOWS, WHERE OCT XXXXX CONTAINS THE PHASE INFORMATION,
060892,000087:                                                                                                 #                 TC      PHASCHNG
060893,000088:                                                                                                 #                 OCT     XXXXX
060894,000089:                                                                                                 #  TYPE A IS CONCERNED WITH FIXED PHASE CHANGES, THAT IS, PHASE INFORMATION THAT IS STORED PERMANENTLY.  THESE
060895,000090:                                                                                                 #  OPTIONS ARE, WHERE G STANDS FOR A GROUP AND .X FOR THE PHASE,
060896,000091:                                                                                                 #         G.0             INACTIVE, WILL NOT PERMIT A GROUP G RESTART
060897,000092:                                                                                                 #         G.1             WILL CAUSE THE LAST DISPLAY TO BE REACTIVATED, USED MAINLY IN MANNED FLIGHTS
060898,000093:                                                                                                 #         G.EVEN          A DOUBLE TABLE RESTART, CAN CAUSE ANY COMBINATION OF TWO JOBS, TASKS, AND/OR
060899,000094:                                                                                                 #                         LONGCALL TO BE RESTARTED.
060900,000095:                                                                                                 #         G.ODD NOT .1    A SINGLE TABLE RESTART, CAN CAUSE EITHER A JOB, TASK, OR LONGCALL RESTART.
060901,000096: 
060902,000097:                                                                                                 #  THIS INFORMATION IS PUT INTO THE OCTAL WORD AFTER TC PHASCHNG AS FOLLOWS
060903,000098:                                                                                                 #         TL0 00P PPP PPP GGG
060904,000099:                                                                                                 #  WHERE EACH LETTER OR NUMBER STANTS FOR A BIT.  THE G'S STAND FOR THE GROUP, OCTAL 1-7, THE P'S FOR THE PHASE,
060905,000100:                                                                                                 #  OCTAL 0 - 127.  0'S MUST BE 0.  IF ONE WISHES TO HAVE THE TBASE OF GROUP G TO BE SET AT THIS TIME,
060906,000101:                                                                                                 #  T IS SET TO 1, OTHERWISE IT IS SET TO 0.  SIMILARLY IF ONE WISHES TO SET LONGBASE, THEN L IS SET TO 1, OTHERWISE
060907,000102:                                                                                                 #  IT IS SET TO 0.  SOME EXAMLES,
060908,000103:                                                                                                 #                 TC      PHASCHNG        # THIS WILL CAUSE GROUP 3 TO BE SET TO 0,
060909,000104:                                                                                                 #                 OCT     00003           # MAKING GROUP 3 INACTIVE
060910,000105: 
060911,000106:                                                                                                 #                 TC      PHASCHNG        # IF A RESTART OCCURS THIS WOULD CAUSE
060912,000107:                                                                                                 #                 OCT     00012           # GROUP 2 TO RESTART THE LAST DISPLAY
060913,000108: 
060914,000109:                                                                                                 #                 TC      PHASCHNG        # THIS SETS THE TBASE OF GROUP 4 AND IN
060915,000110:                                                                                                 #                 OCT     40064           # CASE OF A RESTART WOULD START UP THE TWO
060916,000111:                                                                                                 #                                         # THINGS LOCATED IN THE DOUBLE 4.6 RESTART
060917,000112:                                                                                                 #                                         # LOCATION.
060918,000113: 
060919,000114:                                                                                                 #                 TC      PHASCHNG        # THIS SETS LONGBASE AND UPON A RESTART
060920,000115:                                                                                                 #                 OCT     20135           # CAUSES 5.13 TO BE RESTARTED (SINCE
060921,000116:                                                                                                 #                                         # LONGBASE WAS SET THIS SINGLE ENTRY
060922,000117:                                                                                                 #                                         # SHOULD BE A LONGCALL)
060923,000118: 
060924,000119:                                                                                                 #                 TC      PHASCHNG        # SINCE BOTH TBASE4 AND LONGBASE ARE SET,
060925,000120:                                                                                                 #                 OCT     60124           # 4.12 SHOULD CONTAIN BOTH A TASK AND A
060926,000121:                                                                                                 #                                         # LONGCALL TO BE RESTARTED
060927,000122: 
060928,000123:                                                                                                 #  TYPE C PHASCHNG CONTAINS THE VARIABLE TYPE OF PHASCHNG INFORMATION.  INSTEAD OF THE INFORMATION BEING IN A
060929,000124:                                                                                                 #  PERMANENT FORM, ONE STORES THE DESIRED RESTART INFORMATION IN A VARIABLE LOCATION.  THE BITS ARE AS FOLLOWS,
060930,000125:                                                                                                 #         TL0 1AD XXX CJW GGG
060931,000126:                                                                                                 #  WHERE EACH LETTER OR NUMBER STANDS FOR A BIT.  THE G'S STAND FOR THE GROUP, OCTAL 1 - 7.  IF THE RESTART IS TO
060932,000127:                                                                                                 #  BE BY WAITLIST, W IS SET TO 1, IF IT IS A JOB, J IS SET TO 1, IF IT IS A LONGCALL, C IS SET TO 1.  ONLY ONE OF
060933,000128:                                                                                                 #  THESE THREE BITS MAY BE SET.  X'S ARE IGNORED, 1 MUST BE 1, AND 0 MUST BE 0.  AGAIN T STANDS FOR THE TBASE,
060934,000129:                                                                                                 #  Page 1407
060935,000130:                                                                                                 #  AND L FOR LONGBASE.  THE BITS A AND D ARE CONCERNED WITH THE VARIABLE INFORMATION.  IF D IS SET TO 1, A PRIORITY
060936,000131:                                                                                                 #  OR DELTA TIME WILL BE READ FROM THE NEXT LOCATION AFTER THE OCTAL INFORMATION., IF THIS IS TO BE INDIRECT, THAT
060937,000132:                                                                                                 #  IS, THE NAME OF A LOCATION CONTAINING THE INFORMATION (DELTA TIME ONLY), THEN THIS IS GIVEN AS THE -GENADR OF
060938,000133:                                                                                                 #  THAT LOCATION WHICH CONTAINS THE DELTA TIME.  IF THE OLD PRIORITY OR DELTA TIME IS TO BE USED, THAT WHICH IS
060939,000134:                                                                                                 #  ALREADY IN THE VARIABLE STORAGE, THEN D IS SET TO 0.  NEXT THE A BIT IS USED.  IF IT IS SET TO 0, THE ADDRESS
060940,000135:                                                                                                 #  THAT WOULD BE RESTARTED DURING A RESTART IS THE NEXT LOCATION AFTER THE PHASE INFORMATION, THAT IS, EITHER
060941,000136:                                                                                                 #  (TC PHASCHNG) +2 OR +3, DEPENDING ON WHETHER D HAD BEEN SET OR NOT.  IF A IS SET TO 1, THEN THE ADDRESS THAT
060942,000137:                                                                                                 #  WOULD BE RESTARTED IS THE 2CADR THAT IS READ FROM THE NEXT TWO LOCATION.  EXAMPLES,
060943,000138:                                                                                                 #         AD      TC      PHASCHNG        # THIS WOULD CAUSE LOCATION AD +3 TO BE
060944,000139:                                                                                                 #         AD+1    OCT     05023           # RESTARTED BY GROUP THREE WITH A PRIORITY
060945,000140:                                                                                                 #         AD+2    OCT     23000           # OF 23.  NOTE UPON RETURNING IT WOULD
060946,000141:                                                                                                 #         AD+3                            # ALSO GO TO AD+3
060947,000142: 
060948,000143:                                                                                                 #         AD      TC      PHASCHNG        # GROUP 1 WOULD CAUSE CALLCALL TO BE
060949,000144:                                                                                                 #         AD+1    OCT     27441           # BE STARTED AS A LONGCALL FROM THE TIME
060950,000145:                                                                                                 #         AD+2    -GENADR DELTIME         # STORED IN LONGBASE (LONGBASE WAS SET) BY
060951,000146:                                                                                                 #         AD+3    2CADR   CALLCALL        # A DELTATIME STORED IN DELTIME.  THE
060952,000147:                                                                                                 #         AD+4                            # BBCON OF THE 2CADR SHOULD CONTAIN THE E
060953,000148:                                                                                                 #         AD+5                            # BANK OF DELTIME.  PHASCHNG RETURNS TO
060954,000149:                                                                                                 #                                         # LOCATION AD+5
060955,000150: 
060956,000151:                                                                                                 #  NOTE THAT IF A VARIABLE PRIORITY IS GIVEN FOR A JOB, THE JOB WILL BE RESTARTED AS A NOVAC IF THE PRIORITY IS
060957,000152:                                                                                                 #  NEGATIVE, AS A FINDVAC IF THE PRIORITY IS POSITIVE.
060958,000153: 
060959,000154:                                                                                                 #  TYPE B PHASCHNG IS A COMBINATION OF VARIABLE AND FIXED PHASE CHANGES.  IT WILL START UP A JOB AS INDICATED
060960,000155:                                                                                                 #  BELOW AND ALSO START UP ONE FIXED RESTART, THAT IS EITHER AN G.1 OR A G.ODD OR THE FIRST ENTRY OF G.EVEN
060961,000156:                                                                                                 #  DOUBLE ENTRY.  THE BIT INFORMATION IS AS FOLLOW,
060962,000157:                                                                                                 #         TL1 DAP PPP PPP GGG
060963,000158:                                                                                                 #  WHERE EACH LETTER OR NUMBER STANDS FOR A BIT.  THE G'S STAND FOR THE GROUP, OCTAL 1 - 7, THE P'S FOR THE FIXED
060964,000159:                                                                                                 #  PHASE INFORMATION, OCTAL 0 - 127.  1 MUST BE 1.  AND AGAIN T STANDS FOR THE TBASE AND L FOR LONGBASE.  D THIS
060965,000160:                                                                                                 #  TIME STANDS ONLY FOR PRIORITY SINCE THIS WILL BE CONSIDERED A JOB, AND IT MUST BE GIVEN DIRECTLY IF GIVEN.
060966,000161:                                                                                                 #  AGAIN A STANDS FOR THE ADDRESS OF THE LOCATION TO BE RESTARTED, 1 IF THE 2CADR IS GIVEN, OR 0 IF IT IS TO BE
060967,000162:                                                                                                 #  THE NEXT LOCATION.  (THE RETURN LOCATION OF PHASCHNG) EXAMPLES,
060968,000163:                                                                                                 #         AD      TC      PHASCHNG        # TBASE IS SET AND A RESTART CAUSE GROUP 3
060969,000164:                                                                                                 #         AD+1    OCT     56043           # TO START THE JOB AJOBAJOB WITH PRIORITY
060970,000165:                                                                                                 #         AD+2    OCT     31000           # 31 AND THE FIRST ENTRY OF 3.4SPOT (WE CAN
060971,000166:                                                                                                 #         AD+3    2CADR   AJOBAJOB        # ASSUME IT IS A TASK SINCE WE SET TBASE3)
060972,000167:                                                                                                 #         AD+4                            # UPON RETURN FROM PHASCHNG CONTROL WOULD
060973,000168:                                                                                                 #         AD+5                            # GO TO AD+5
060974,000169: 
060975,000170:                                                                                                 #         AD      TC      PHASCHNG        # UPON A RESTART THE LAST DISPLAY WOULD BE
060976,000171:                                                                                                 #         AD+1    OCT     10015           # RESTARTED AND A JOB WITH THE PREVIOUSLY
060977,000172:                                                                                                 #         AD+2                            # STORED PRIORITY WOULD BE BEGUN AT AD+2
060978,000173:                                                                                                 #                                         # BY MEANS OF GROUP 5
060979,000174:                                                                                                 #  Page 1408
060980,000175:                                                                                                 #  THE NOVAC-FINDVAC CHOICE FOR JOBS HOLDS HERE ALSO -- NEGATIVE PRIORITY CAUSES A NOVAC CALL, POSITIVE A FINDVAC.

060988,000183:                                                                                                 #  2PHSCHNG IS USED WHEN ONE WISHES TO START UP A GROUP OR CHANGE A GROUP WHILE UNDER THE CONTROL OF A DIFFERENT
060989,000184:                                                                                                 #  GROUP.  FOR EXAMPLE, CHANGE THE PHASE OF GROUP 3 WHILE THE PORTION OF THE PROGRAM IS UNDER GROUP 5.  ALL 2PHSCHNG
060990,000185:                                                                                                 #  CALLS ARE MADE IN THE FOLLOWING MANNER,
060991,000186:                                                                                                 #                 TC      2PHSCHNG
060992,000187:                                                                                                 #                 OCT     XXXXX
060993,000188:                                                                                                 #                 OCT     YYYYY
060994,000189:                                                                                                 #  WHERE OCT XXXXX MUST BE OF TYPE A AND OCT YYYYY MAY BE OF EITHER TYPE A OR TYPE B OR TYPE C.  THERE IS ONE
060995,000190:                                                                                                 #  DIFFERENCE --- NOTE: IF LONGBASE IS TO BE SET THIS INFORMATION IS GIVEN IN THE OCT YYYYY INFORMATION, IT WILL
060996,000191:                                                                                                 #  BE DISREGARDED IF GIVEN WITH THE OCT XXXXX INFORMATION.  A COUPLE OF EXAMPLES MAY HELP,
060997,000192:                                                                                                 #         AD      TC      2PHACHNG        # SET TBASE3 AND IF A RESTART OCCURS START
060998,000193:                                                                                                 #         AD+1    OCT     40083           # THE TWO ENTRIES IN 3.8 TABLE LOCATION
060999,000194:                                                                                                 #         AD+2    OCT     05025           # THIS IS OF TYPE C, SET THE JOB TO BE
