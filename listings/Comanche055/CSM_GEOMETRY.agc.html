012660,000302:                                                                                                 #  SUBROUTINE NAME:  EARTH ROTATOR        (EARROT1 OR EARROT2)            DATE:  15 FEB 67
012661,000303:                                                                                                 #  MOD NO:  N +1                                                          LOG SECTION:  POWERED FLIGHT SUBROS
012662,000304:                                                                                                 #  MOD BY:  ENTRY GROUP (BAIRNSFATHER)
012663,000305: 
012664,000306:                                                                                                 #  FUNCTIONAL DESCRIPTION:  THIS ROUTINE PROJECTS THE INITIAL EARTH TARGET VECTOR RTINIT AHEAD THROUGH
012665,000307:                                                                                                 #         THE ESTIMATED TIME OF FLIGHT.  INITIAL CALL RESOLVES THE INITIAL TARGET VECTOR RTINIT INTO EASTERLY
012666,000308:                                                                                                 #         AND NORMAL COMPONENTS RTEAST AND RTNORM.  INITIAL AND SUBSEQUENT CALLS ROTATE THIS VECTOR
012667,000309:                                                                                                 #         ABOUT THE (FULL) UNIT POLAR AXIS UNITW THROUGH THE ANGLE  WIE DTEAROT  TO OBTAIN THE ROTATED
012668,000310:                                                                                                 #         TARGET VECTOR RT.  ALL VECTORS EXCEPT UNITW ARE HALF UNIT.
012669,000311:                                                                                                 #         THE EQUATIONS ARE:
012670,000312:                                                                                                 #                 _    _        _                      _
012671,000313:                                                                                                 #                 RT = RTINIT + RTNORM (COS(WT) - 1) + RTEAST SIN(WT)
012672,000314: 
012673,000315:                                                                                                 #         WHERE   WT = WIE DTEAROT
012674,000316: 
012675,000317:                                                                                                 #                 RTINIT = INITIAL TARGET VECTOR
012676,000318:                                                                                                 #                 _        _       _
012677,000319:                                                                                                 #                 RTEAST = UNITW * RTINIT
012678,000320:                                                                                                 #                 _        _        _
012679,000321:                                                                                                 #                 RTNORM = RTEAST * UNITW
012680,000322: 
012681,000323:                                                                                                 #         FOR CONTINUOUS UPDATING, ONLY ONE ENTRY TO EARROT1 IS REQUIRED, WITH SUBSEQUENT ENTRIES AT EARROT2.
